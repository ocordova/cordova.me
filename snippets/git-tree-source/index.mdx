import { getImage, GatsbyImage } from 'gatsby-plugin-image'

## Create alias

```bash
git config --global alias.tree '! git ls-tree --full-name --name-only -t -r HEAD | sed -e "s/[^-][^\/]*\//   |/g" -e "s/|\([^ ]\)/|-- \1/"'
```

## Usage

```bash
git tree
```
